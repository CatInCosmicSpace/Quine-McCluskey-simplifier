723 auto Quine_McCluskey_Simplifier::string_base10_to_base2(std::string inp) const -> std::string {

729  for (size_t i = 0; i < inp.length(); ++i) {

736  }

738  inp = right;
739  } while (inp != std::string());
740 
741  for (size_t i = 0; i < res.length() / 2; ++i)
742  std::swap(res[i], res[res.length() - 1 - i]);

836 log_expr::log_expr(const std::string& str) {
837  std::stringstream ss(str);
838  root_ = new exp_node(nullptr, e_root);
839  exp_node* cur = root_;
840  size_t i;
841  for (i = 0; ss; ++i) {
842  char c = ss.get();
843  if (!ss)
844  break;
845 
846  switch (c) {
847  case '&':
848  case '+': {
849  op_type t;
850  if (c == '&')
851  t = op_and;
852  else
853  t = op_or;
854  if ((cur->state() == 0) || (cur->state() == 2))
855  throw std::runtime_error("Variable missing at position " + std::to_string(i));
856  if (cur->state() == 1)
857  cur->operation = t;
858  else if (cur->state() == 3) {
859  if (t == op_or) {
860  exp_node* temp = new exp_node(*cur);
861  cur->first = temp;
862  temp->prev = cur;
863  cur->operation = t;
864  cur->second = nullptr;
865  }
866  else {
867  exp_node* temp = new exp_node(cur, e_leaf, cur->second->data);
868  cur = cur->second;
869  cur->first = temp;
870  cur->operation = t;
871  cur->type = e_node;
872  }
873  }
874  break;
875  }
876  case '!': {
877  if ((cur->state() == 1) || (cur->state() == 3))
878  throw std::runtime_error("Operator missing at position " + std::to_string(i));
879  if (cur->operation == op_not)
880  throw std::runtime_error("Double negage at position " + std::to_string(i));
881  exp_node* temp = new exp_node(cur, e_node);
882  temp->operation = op_not;
883  if (cur->state() == 0) {
884  cur->first = temp;
885  cur = cur->first;
886  }
887  else if (cur->state() == 2) {
888  cur->second = temp;
889  cur = cur->second;
890  }
891  break;
892  }
893  case '(': {
894  if ((cur->state() == 1) || (cur->state() == 3))
895  throw std::runtime_error("Operator missing at position " + std::to_string(i));
896  exp_node* temp = new exp_node(cur, e_subnode);
897  if (cur->state() == 0) {
898  cur->first = temp;
899  cur = cur->first;
900  }
901  else if (cur->state() == 2) {
902  cur->second = temp;
903  cur = cur->second;
904  }
905  break;
906  }
907  case ')': {
908  while ((cur->type != e_subnode) && (cur->type != e_root))
909  cur = cur->prev;
910  if (cur->type == e_root)
911  throw std::runtime_error("Left bracket not found at position " + std::to_string(i));
912  cur = cur->prev;
913  if (cur->operation == op_not)
914  cur = cur->prev;
915  break;
916  }
917  case ' ':
918  case '\t':
919  break;
920  default: {
921  std::string id(1, c);
922  while (ss) {
923  c = ss.get();
924  if (!ss)
925  break;
926  if ((c < '0') || (c > '9')) {
927  ss.unget();
928  break;
929  }
930  id += c;
931  ++i;
